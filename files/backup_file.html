function backup_file() { local timestamp=$(date +%Y%m%d%H%M%S) local backup_filename="${filename##*/}.backup.${timestamp}" # Extract filename only local backup_path="${backup_dir}/${backup_filename}" if [ -f "$filename" ]; then if [ -d "$backup_dir" ]; then # Check if backup directory exists cp "$filename" "$backup_path" echo "File '$filename' backed up to '$backup_path'." else echo "Error: Backup directory '$backup_dir' does not exist." return 1 fi else echo "Error: File '$filename' not found." return 1 fi }
